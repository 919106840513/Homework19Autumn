#include<iostream>
#include<string.h>
using namespace std;
int getRev(char a[])
{
	int length = strlen(a);
	int count = 0;
	for(int i=0;i<length;i++)
	{
		for(int j=i+1;j<length;j++)
		{
			if(a[i]>a[j])
				count++;
		}
	}
	return count;
}
int main()
{
	int max;
	cout<<"请输入需要计算的字符串的个数："<<endl;
	cin>>max;
	cout<<"请输入长度不大于20的字符串："<<endl;
	char a[max][20];
	int num[max];
	for(int i=0;i<=max;i++)
	{
		cin.getline(a[i],20); 
		num[i] = getRev(a[i]); 
	}
	for(int i=0;i<=max-1;i++)
	{
		int temp;
		char temp_a[20];
		for(int j=i+1;j<=max;j++)
		{
			if(num[i]>num[j])
			{
				temp = num[i];
				strcpy(temp_a,a[i]);
				num[i] = num[j];
				strcpy(a[i],a[j]);
				num[j] = temp;
				strcpy(a[j],temp_a);			
			}
		} 
	} 
	cout<<"结果为："<<endl;
	for(int i=0;i<=max;i++)
	{
		cout<<a[i]<<" "<<num[i]<<endl;
	}	
	return 0;
}

#include<iostream>
#include<string.h>
using namespace std;
void getReapt(string s)
{
	int count;
	for(int i=0;i<s.size();i++)
	{
		count = 1;
		for(int j=0;j<i;j++)
		{
			if(s[i]==s[j])
			{
				i++;
				j=0;
			}
		}
		for(int j=i+1;j<s.size();j++)
		{
			if(s[i]==s[j])
				count++;
		}
		if(s[i] != '\0')
			cout<<s[i]<<":"<<count<<endl;
	}
}
int main()
{
	int n;
	cout<<"请输入字符串个数："<<endl;
	cin>>n;
	string str[n];
	int num[n];
	cout<<"请输入字符串："<<endl;
	for(int i=0;i<n;i++)
	{
		cin>>str[i];
	}
	for(int i=0;i<n;i++)
	{
		cout<<"第"<<i+1<<"个字符串中字符出现的次数依次是："<<endl;
		getReapt(str[i]);	
	}
	return 0;
}
