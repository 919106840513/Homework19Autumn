#include<iomanip>
using namespace std; 
int yueshu(int a,int b){
   while (a%b != 0) 
     {
        int temp = a;
        a = b;
        b = temp % a;
     }
    return b;
}
int beishu(int a,int b){
	int x=yueshu(a,b);
	return a*b/x;
}

#include<iostream>
#include<iomanip>
using namespace std;
int daoxu(int x){
	while(x!=0){
		cout<<x%10;
		x/=10;
	}
}
int main(){
	int x;
	cout<<"请输入一个整数: ";
	cin>>x;
	daoxu(x);
	return 0;
}
