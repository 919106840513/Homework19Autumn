#include <string.h>
#include <iostream>
using namespace std;
int getRev(char *a){
	char *t=a;
	int count=0,n=0;
    do{
		for(int j=1;j<strlen(a)-n;j++){
			if((int)*t>(int)*(t+j))
			count++;
		}
		t++;
		n++;
	}while(strlen(a)-n!=1);
	return count;
}
int main(){
	char a[20],b[20],c[20],d[20];
	int g[4];
	int temp;
	cin>>a>>b>>c>>d;
    g[0]=getRev(a);
    g[1]=getRev(b);
    g[2]=getRev(c);
    g[3]=getRev(d);
 	for(int i=0;i<3;i++) {
		for(int j=i+1; j<4;j++)	{
			if(g[i]>g[j]){
				temp=g[i]; 
				g[i]=g[j]; 
				g[j]=temp;
			}
}}
	for(int i=0;i<4;i++){
		cout<<g[i]<<" ";
	}
}

#include <string.h>
#include <iostream>
using namespace std;
int getWorldCount(char *a){
	char *t=a;
	int count=0;
	do{
	if(((int)*t==32||(int)*t==9)&&((int)*(t+1)!=32||(int)*(t+1)!=9))
	count++;
	t++;
	}while(*t!=NULL);
	return count+1;
}
