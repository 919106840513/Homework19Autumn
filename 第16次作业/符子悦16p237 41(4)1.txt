using namespace std;
#include <iostream>
#include <string.h>
void ad(char*pb[],int m){
	char *temp;
	for(int f=1;f<m;f++){
		for(int f1=f+1;f1<=m;f1++){
		
	if(strcmp(pb[f],pb[f+1])>0){
	temp=pb[f];
	pb[f]=pb[f1];
	pb[f1]=temp;	
	   }}
	}
	for(int d=1;d<=m;d++){
		cout<<pb[d]<<endl;
   }
}
void ap(char *as[],int m){
	char d[m][51],p[m][51];
	int i,u,k,k1,p1[m];
	for(i=0;i<m;i++){
	p1[i]=1;	
	}
    for(i=1;i<=m;i++){
    strcpy(d[i],as[i]);
	    }
	for(i=1;i<=m;i++){
		for(u=i-1;u>=1;u--){
	if(strcmp(p[u],d[i])==0)break;}
	k1=1;strcpy(p[k1],d[i]);
	for(k=i+1;k<=m;k++){
	if(strcmp(d[i],d[k])==0)p1[k1-1]++;	
	  }k1++;  	
	}
	for(i=1;i<=m;i++){
	cout<<p[i]<<" "<<p1[i-1]<<endl;	
	}    
}
int main (){
