using namespace std;
#include<iostream>
#include<string.h>
int getWordCount(char a[])
{
	int count=1;
	for(;*a!=0;)
	{
		if((*a==32||*a==9)&&(*(a+1)!=32&&*(a+1)!=9))
		count++;
		a++;
	}
	return count;
}

int main()
{
	char a[1000];
	cin.getline(a,1000);
	cout<<getWordCount(a);
	return 0;
}
