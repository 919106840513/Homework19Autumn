#include<iostream>
using namespace std;
char *strcat(char *a,char *b)
{
	if(a!=NULL&&b!=NULL)
	{
		char *t=a;
		while(*a)
			a++;
		for(;*b!=0;)
		{
			*a=*b;
			a++;
			b++;
		}
		*a=0;
		return t;
	}
}
int main()
{
	char s1[10]="week", s2[20]={"day"};
	cout<<strcat(s1,s2)<<endl; 
	cout<<s1<<" "<<s2;
	return 0;
}





#include<iostream>
using namespace std;
char * trim(char * s)
{
	char *t=s;
	char *t1=s;
	for(;*s!=0;)
		{
			if(*s==32)
			{
				t1=s;
				for(;*s!=0;)
				{
					*s=*(s+1);
					s++;
				}
				s=t1;
			}
			else
			s++;			
		}
		return t;
}
int main()
{
	char test[10000];
	char *str=test;
	cin.getline(str,100);
	cout<<trim(test);
	return 0;	
}
