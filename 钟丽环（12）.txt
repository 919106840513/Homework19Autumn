#include <iostream>
using namespace std;
int getSet(const int a[],int n){
	int rs[100];
	int i,j,k=0;
	for(i=0;i<n;i++){
		if(i==0)
			k++;
		else
		for(j=0;j<i;j++){
			if(a[i]==a[j])
			break;
		}
		if(j==i)
		k++;
	}
	return k;
}
int Z(int a[],int n){
	int i,k,l,j;
	cout<<a[0]<<" ";
	for(i=1;i<n;i++){
			j=0;
		do{
			if(a[i]==a[j])
			break;
			else
			j++;
		}while(j<i);
		if(j=i){
			cout<<a[i]<<" ";
		for(k=i,l=0;k<n;k++){
			if(a[i]=a[k])
			l++;
		}cout<<l<<" ";
	}
	}
}

int main(){
