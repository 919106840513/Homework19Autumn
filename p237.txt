int getRev(char *a)
{
	char temp;
	int count=0;
	char *p=a;
	char *q=a;
	for(;*(p+1)!=0;p++)
	{
		a=q;
		for(;*(a+1)!=0;a++)
		{
			if(*a>*(a+1))
			{
				temp=*a;
				*a=*(a+1);
				*(a+1)=temp;
				count++;
			}			
		}	
	}
	return count;	
}
