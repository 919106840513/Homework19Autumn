11.4using namespace std;
#include <iostream>
int main()
{
	int count=0;
	char ch=0;
	do	{
	ch=cin.get();
	if (ch>='1'&&ch<='9')
	count ++;
	}while(ch!='\n');
	cout<<"count="<<count<<endl;
	return 0;
}
