#include<iostream>
#include<math.h>
using namespace std;
int main(){
	float a,b,c,m,x1,x2;
	cin>>a>>b>>c;
	m=b*b-4*a*c;
	if(m<0){
		cout<<"方程无根"<<endl; 
	}
	else if(m==0){
		x1=-b/(2*a);
		cout<<x1<<endl;
	}
	else{
		m=sqrt(m);
		x1=(-b+m)/(2*a);
		x2=(-b-m)/(2*a);
		cout<<x1<<x2<<endl;
	}
	return 0;
}
