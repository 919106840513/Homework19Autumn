#include <iostream>
#include <math.h>
using namespace std;
int main(void)
{
 int a,b,c,d;
 cout<<"求方程a*x*x+b*x+c=0的根，输入a,b,c";
 cin>>a>>b>>c;
 d=b*b-4a*c;
 if(d>0)
 {
  cout<<"x1=(-b+sqrt(d))／(a*2)";
  cout<<"x2=(-b-sqrt(d))／(a*2)"<<endl;
 }
 else if(d==0)
 cout<<"x1=x2=-b／(a*2)"<<endl;
 else
 cout<<"方程无解"<<endl;
 return 0;
}
