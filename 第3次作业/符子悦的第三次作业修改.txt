11（1）
using namespace std;
#include <iostream>
int main (void)
{
int a,b=0,c=1;
cout<<"输入一个整数"<<endl;
cin>>a;
while(a>0)
{b=(a%2)*c+b;a=a/2;c=c*10;}
cout<<"其二进制为"<<b;
return 0;
}
