cout<<"该方程无根"<<endl;
else if(b*b-4*a*c==0)
{x=-b/(2*a);
cout<<"该方程有一根"<<x<<endl;}
else
{ (b*b-4*a*c>0);
x=-b-(sqrt(b*b-4*a*c))/(2*a), d=-b+(sqrt(b*b-4*a*c))/(2*a);
