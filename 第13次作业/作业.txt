#include <iostream>
#include <string.h> 
using namespace std;
int getRev(char a[]){	
	int k=0,s;  	 
	s=strlen(a);		
	for(int i=0;i<s;i++)
	{
		   	for(int j=i+1;j<s;j++)
		   {		   
	   		if(a[i]>a[j])
	   		    k++;
		   }		   	   
	} 	
   return k;	
}

void bubbleSort(int data[], int n){
	int i, j, temp;
	for(i = 0; i < n-1; i++)//外层循环控制总轮次
		for(j = 0; j < n-1-i; j++)//内层循环比较次数
			if(data[j] > data[j+1]){	//是否要交换
				temp = data[j]; 
				data[j] = data[j+1]; 
				data[j+1] = temp;
			}
}

#include <iostream>
#include <string.h> 
using namespace std;
int getWordCount(char a[]){
	int count=0,l;
	l=strlen(a);
	for(int i=0;i<l;i++)
	{
	  if(a[i]!=' '&&a[i]!='	')
	     count+=0;
	     else count++;
	}
	count+=1;
	cout<<count;
}

int main(){
	char a[100];
	cin.getline(a,100);	
	getWordCount(a);
}
