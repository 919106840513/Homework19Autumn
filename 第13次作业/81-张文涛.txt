#include <iostream>
#include <string.h>
#define MAX 2
#define size 20
using namespace std;
int getRev(char a[])
{
	int i,j,count=0,temp;
	int length=strlen(a);
	for(i=0;i<length;i++)
	{
		for(j=1;j<length;j++)
		{
			if(i<j&&a[i]>a[j])
			{
				count++;
				temp=a[i]; 
				a[i]=a[j]; 
				a[j]=temp;
			}
			
			else continue;
		}
	}cout<<a<<" ";
	return count;
}

int main()
{
	char a[]={'\0'};
	gets(a);
	cout<<getRev(a);
}

#include <iostream>
#include <string.h>
using namespace std;
int getWordCount(char a[])
{
	int count=0,i=0;
	int length=strlen(a);
	for(int i=0;i<=length;i++)
	{
		if(a[i]!=' '&&a[i]!='\t')
        {
            count++;
            while(a[i]!=' '&&a[i]!='\0'&&a[i]!='\t')
            {
                i++;
            }
        }
	}
   	return count;
}

int main()
{
	char a[100];
	cin.getline(a,100);
	cout<<getWordCount(a);
	return 0;
}
