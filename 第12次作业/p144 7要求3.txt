using namespace std;
#include<iostream>

void timeappearorder(const int a[], int n0)
{
	int max=a[0];
	for(int i=1;i<n0;i++)
	{
		if(a[i]>max)
		max=a[i]; 
	}
	int digit=max+1;
	//cout<<digit<<endl;
	int store[digit];
	for(int i=0;i<digit;i++)
	{
		store[i]=0;
	}
	for(int i=0;i<n0;i++)
	{
		int present=a[i];
		store[present]++;
	}
	int index[digit];
	for(int i=0;i<digit;i++)
	{
		index[i]=i;
	}
	int i, j;
	for(j=1;j<digit;j++)
	{
		for(i=1;i<digit-j;i++)
		{
			if(store[i+1]>store[i])
			{
				int tempstore=store[i];
				store[i]=store[i+1];
				store[i+1]=tempstore;
				int tempindex=index[i];
				index[i]=index[i+1];
				index[i+1]=tempindex;
			}
		}
	}
		
	
	int outputorder;
	for(int i=1;i<digit;i++)
	{
		outputorder=index[i];
		cout<<outputorder<<":"<<store[i]<<endl;
	}

}	

int main()
{
	int n;
	cin>>n;
	int a[n];
	int i;
	for(i=0;i<n;i++)
		cin>>a[i];

 timeappearorder(a, n);	
	
	return 0;
}
