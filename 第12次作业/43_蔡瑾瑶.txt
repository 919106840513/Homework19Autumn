int getxs(int rs[],int cs[],const int a[],int n){
	int r=1;
	rs[0]=a[0];
	for(int i=1;i<n;i++){
		for(int j=0;j<i;j++){
			if(a[j]==a[i])
		   break;
			if(j==i-1)
		   	rs[r++]=a[i];
		} 
	}
	for(int i=0;i<r;i++){
		for(int j=0;j<n;j++){
			if(rs[i]==a[j])
			cs[i]++;
		}	   
	}
	int h,t;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++)

		   	h=cs[i];cs[i]=cs[j];cs[j]=h;
		   	t=rs[i];rs[i]=rs[j];rs[j]=t;
		   }  
	}

	cout<<rs[i]<<":"<<cs[i]<<endl; 
	return r;
}
