int getSet(int rs[],const int a[],int n)
{
int count=0;int k=1;int sum=1;
rs[0]=a[0];
for(int i=1;i<n;i++)
{count=0;
for(int j=0;j<k;j++)
{
if(a[i]!=rs[j])
count++;
}
if(count==k)
{rs[k]=a[i];
k++;
sum++
}
}
return sum;
}

int COUNT(int rs[],int num;const int a[],int n)
{
int sum=0;
for(int r=0;r<n;r++)
{if(rs[num]==a[r])
sum++;
}return sum;
}

int Paixu(int a[],int n)
{
int i,j,min,Index,tmp;
for(j=i+1;j<n;j++)
{
if(min>a[j])
{
min=a[j];
Index=j;
}
}
if(i!=Index)
{
tmp=a[i];
a[i]=a[Index];
a[Index]=tmp;
}
}
}
