函数
#include <iostream>
#include <math.h>
using namespace std;
double area(double a, double b, double c) {
	double s = (a + b + c) / 2;
	return sqrt(s * (s - a) * (s - b) * (s - c));
}
int main() {
	double a, b, c, d;
	cin >> a >> b >> c;
	d = area(a, b, c);
	cout << d ;
}
