#include<math.h>
#define S(a,b,c) (a+b+c)/2
#define Area(s,a,b,c)  sqrt(s*(s-a)*(s-b)*(s-c))
using namespace std;
int main(){
	double a,b,c,s;
	cout<<"请输入三角形三边长：";
	cin>>a>>b>>c;
	s=S(a,b,c);
	cout<<"三角形的面积为："<<Area(s,a,b,c)<<endl;
    return 0;
}
函数：
#include<iostream> 
#include<math.h>
using namespace std;
double area(int a,int b,int c,int s){
	return sqrt(s*(s-a)*(s-b)*(s-c));
}
int main(){
	double a,b,c,s;
	cout<<"请输入三角形三边长：";
	cin>>a>>b>>c;
	s=(a+b+c)/2;
	cout<<"三角形的面积为："<<area(a,b,c,s)<<endl;
    return 0;
}
