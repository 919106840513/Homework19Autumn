21-（9）函数
#include<iostream>
#include<math.h>
using namespace std;
bool is(float a,float b,float c)
{
	if(a<=0||b<=0||c<=0) return false;
	if(a+b<=c||a+c<=b||b+c<=a) return false;
	return true;
}
float sum(float a,float b,float c)
{
	float s;
	s=(a+b+c)/2;
	float sum;
	sum=sqrt(s*(s-a)*(s-b)*(s-c));
	return sum;
}
int main()
{
	float a,b,c;
	cin>>a>>b>>c;
	while(!is(a,b,c))
	{
		cout<<"非三角形,请输入正确的三条边长："<<endl;
		cin>>a>>b>>c;
	}
	cout<<"面积为："<<sum(a,b,c)<<endl;
}
