//1.指针实现strcat函数，不用string.h
#include<iostream>
using namespace std;
char *strcat(char *str1,char *str2){
	char *p=str1;
	while((*str1)!='\0')
		str1++;
	while((*str2)!='\0'){
		*str1++=*str2++;
	}
	*str1='\0';
	return p;
}
int main(){
	char str1[100];
	char str2[100];
	gets(str1);
	gets(str2);
	cout<<strcat(str1,str2)<<endl;
	return 0;
} 


//2.去除字符串空格
#include<iostream>
#include<string.h>
using namespace std;
char *trim(char *s){
	if(s==NULL)	return NULL;
	if(strlen(s)==0)	return s;
	char *p=s;
	while(*p!=0){
		if(*p==' '){
			char *q=p;
			while(*q!=0){
				*q=*(q+1);
				q++;
			}
		}
		else	p++;
	}
	return s;
}
int main(){
	char s[]=" ab c 1 2 3 ";
	cout<<trim(s)<<endl;
	return 0;
}
