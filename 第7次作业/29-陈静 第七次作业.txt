  if(num1<num2){

    temp=num1;

    num1=num2;

    num2=temp;

  }

  a=num1;b=num2;

  while(b!=0){

    temp=a%b;

    a=b;

    b=temp;

  }

#include<iostream>

using namespace std;

int main(){

	int m,n;

	cout<<"请依次输入底边长度和高度：";

	cin>>m>>n;

	for(int i=0;i<n;i++){

		for(int k=0;k<i;k++)

			cout<<" ";

		for(int j=0;j<m;j++)

			cout<<"*";

		cout<<endl;	

	}

	cout<<endl;

	for(int i=0;i<n;i++){

		for(int k=n-1;k>i;k--)

			cout<<" ";

		for(int j=0;j<m;j++)

			cout<<"*";

		cout<<endl;

	}	

	return 0;

}
