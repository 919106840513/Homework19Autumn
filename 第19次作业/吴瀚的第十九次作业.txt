#include <iostream>
using namespace std;
int getRev(char *a){	
	int k=0;  	 		
	for(int i=0;*(a+i);i++)
	{
		   	for(int j=i+1;*(a+j);j++)
		   {		   
	   		if(a[i]>a[j])
	   		    k++;
		   }		   	   
	} 	
   return k;	
}

void bubbleSort(int *data, int n){
	int i, j, temp;
	for(i = 0; i < n-1; i++)
		for(j = 0; j < n-1-i; j++)
			if(*(data+j) > *(data+j+1)){	
				temp = *(data+j); 
				*(data+j) = *(data+j+1); 
				*(data+j+1) = temp;
			}
}

#include <iostream>
#define M 100
using namespace std;
int getWordCount(char *a){
	int count=0;
	for(int i=0;*(a+i);i++)
	{
	  if(*(a+i)!=' '&&*(a+i)!='	')
	     count+=0;
	     else count++;
	}
	count+=1;
	return count;
}

int main(){
	int n;
	char a[M];
	cin.getline(a,100);	
	n=getWordCount(a);
	cout<<"The number of the words is "<<n;
}
