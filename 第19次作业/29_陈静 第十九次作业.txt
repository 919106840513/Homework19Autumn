  int getWordCount(char* a)



 {

 	{

 			key=1;



 			n++;



 		}



 		if(*(a+x)>='A'&&*(a+x)<='Z'&&key==0)



 		{



 			key=1;



 			n++;



 		}



 		if((*(a+x)=='\t'||*(a+x)==' ')&&key==1)		



 			key=0;		



 	}



 	return n;



 }
