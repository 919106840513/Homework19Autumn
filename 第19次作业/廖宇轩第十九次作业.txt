（6）#include <iostream>
#include <string.h>
#define MAX 3
using namespace std;

int getRev(char a[])
{
	int s = 0;
	int len = strlen(a);
	for (int i = 0; a[i]; i++)
	{
		for (int j = i + 1; a[j]; j++)
		{
			if ((int)a[i] > (int)a[j])
			{
				s++;


			}


		}


	}
	return s;

}


int main()
{
	char a[21 * MAX];
	int s;
	cin.getline(a, 21);
	for (int i = 0; i < 21 * MAX; i++)
	{
		if (a[i] == ' ')
		{
			a[i] = '\0';

		}

	}
	int len[MAX] = { 0 };
	int b[MAX];
	for (int i = 0; i < MAX; i++)
	{

		*(b+i) = getRev(a + *(len + i));
		if (i != MAX - 1)
		{
			*(len+i+1) = strlen(a + *(len + i)) + *(len + i) + 1;
		}

	}

	for (int i = 0; i < MAX; i++)
	{
		for (int j = 0; j < MAX - 1 - i; j++)
		{
			if (b[j] > b[j + 1])
			{
				s = b[j];
				b[j] = b[j + 1];
				b[j + 1] = s;
				s = len[j];
				len[j] = len[j + 1];
				len[j + 1] = s;
			}


		}
	}
	for (int i = 0; i < MAX; i++)
	{
		cout << a + len[i] << '\t' << b[i] << endl;
	}

}

（8）#include <iostream>
#include <string.h>
using namespace std;

int isword(char x)
{
	if ((x <= 'z'&&x >= 'a') || (x <= 'Z'&&x >= 'A'))
	{
		return 1;

	}

}

int getWordCount(char a[])
{
	int s = 0;
	for (int i = 0; i <= strlen(a); i++)
	{

		if (isword(*(a + i)) && (*(a + i + 1) == ' ' || *(a + i + 1) == '\t' || *(a + i + 1) == '\0'))
		{
			s++;

		}
		
	}
	return s;
}
int main()
{
	char a[100];
	cin.getline(a, 100);
	cout << getWordCount(a);

}
