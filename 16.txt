#include <iostream>
#include <cstring>
using namespace std;
void csort(char * c[],int n){
	char * temp;
	for(int i=0;i<n-1;++i){
		for(int j=i;j<n-1;++j){
			if(strcmp(c[j],c[j+1])>0)
			temp=c[j];
			c[j]=c[j+1];
			c[j+1]=temp;
		}
	}
}
void show(char * c[],int n){
	for(int i=0;i<n;++i)
	cout<<c[i]<<endl;
}

int main()
{
	int n;
	char c[20];
	char * c1[10];
	cin>>n;
	getchar();
	for(int i=0;i<n;++i){
		cin.getline(c[i],20);
		c1[i]=new char (strlen(c[i]));
		c1[i]=c[i];
	}
	csort(c1,n);
	show(c1,n);
	return 0;
}

#include <iostream>
#include <cstring>
using namespace std;
void f(char c[],int t[],int &a){
	int co=0;
	for(int i=strlen(c)-1;i>=0;--i){
		int s=0,k=1;
		while(c[i]>='0'&&c[i]<='9'&&i>=0){
			s+=k*(c[i]-'0');
			k*=10;
			i--;
		}
		if(s!=0)
		t[co++]=s;
	}
	a=co;
	return;
}
int main()
{
	int n,a;
	int t[100];
	char c[100];
	cin.getline(c,100);
	f(c,t,a);
	cout<<"个数为："<<a<<endl;
	for(int i=a-1;i>=0;--i)
	cout<<t[i]<<endl;
	return 0;
}
