{   char*temp;

　　　　	{  temp=strs[j];
　　　　　　	strs[j]=strs[j+1];
　　　　　　	strs[j+1]=strs[j];
　　　　　　	}

}

#include<iostream>
#include<cstring>
using namespace std;  //实现指针数组的输入 
static int count=0;
int main(){
	int n;
	string b[100];   //字符串数组 
	cin>>n;
	/*char*strings[100];   //声明 */   //野指针
	char *strings[100];
	char a[100][100];   //使其不为野指针 
	for(int i=0;i<=n;i++)   //字符串结束标志   '0' 
	{
		cin.getline(a[i],100);     //*****输入第i行元素****二维数组中a[i]同样可用 
		strings[i]=a[i];
	}   //输入完 n个字符数组
	for(int i=0;i<=n;i++)
	{
		for(int j=0;j<20;j++)
		{
			if((strings[i][j]>'1')&&(strings[i][j]<'9')&&(strings[i][j+1]>'1')&&(strings[i][j+1]<'9'))  //判断是否连续 
			{
				int num=j;
				char c[100];   //
				for(;((strings[i][num]>'1')&&(strings[i][num]<'9')) ;num++)   //是数字继续循环 
				{
					 c[num-j]=strings[i][num];
					string b[count]=(string)c;   //强制类型转换 
				} 
				
				count++; 		 
			}
		}
    } 
	 for(int i=0;i<=count;i++)
	 {
	 	cout<<b[count]<<endl;
	  } 
	  cout<<count;
	  return 0; 
}
