#include<iostream>
using namespace std;
void transform(char*str)
{
	for(int i=0;*(str+i);i++)
	{
		if(*(str+i)<='Z'&&*(str+i)>='A')
		{
			*(str+i)=*(str+i)+('a'-'A');
		}
	}
	for(int i=0;*(str+i);i++)
		cout<<*(str+i);	
}
int main()
{
	char a[100];
	cin.getline(a,100);
	transform(a);
	return 0;
}


2.#include<iostream>
#include<cstring>
using namespace std;
static int count=0;
void bubblesort(char*strs[],int n)
{   char*temp;    
	for(int i=0;i<=n-1;i++)
	{
		for(int j=0;j<=n-i-1;j++)
		{
		   if(strcmp(strs[j],strs[j+1])>0)
				{temp=strs[j];
				strs[j]=strs[j+1];
				strs[j+1]=temp;
				count++;
				}
		}
	}

}
int main()					//也可用二维数组存放一组字符串    //string 对象输入忽略开头遇到的所有空字符，从第一个有效字符开始输入，直到遇到下一个空字符结束。 
{
	/*string str;
	getline(cin,str);  //输入*/ 
	int n;cin>>n;int num=n;
	char*strings[100];
	char a[100][100];   //使其不为野指针 
	for(int i=0;i<=n;i++)   //字符串结束标志   '0' 
	{
		cin.getline(a[i],100);     //*****输入第i行元素****二维数组中a[i]同样可用 
		strings[i]=a[i];
	}
	bubblesort(strings,num);
	for(int i=0;i<=num;i++)
	{
		cout<<strings[i]<<endl;  //输出 
	}
	cout<<count;
}
