using namespace std;
#include<iostream>
#define Max 5
int getRev(char *a)
{
	char temp;
	int count=0;
	char *p=a;
	char *q=a;
	for(;*(p+1)!=0;p++)
	{
		a=q;
		for(;*(a+1)!=0;a++)
		{
			if(*a>*(a+1))
			{
				temp=*a;
				*a=*(a+1);
				*(a+1)=temp;
				count++;
			}			
		}	
	}
	return count;	
}
int main()
{
	char in[Max][20];
	for(int i=0;i<Max;i++)
		cin.getline(in[i],20);
			
	int rev[Max];
	for(int i=0;i<Max;i++)
		rev[i]=getRev(in[i]);
		
	int index[Max];
	for(int i=0;i<Max;i++)
		index[i]=i;	
	
	for(int i=0;i<Max-1;i++)
	{
		for(int j=0;j<Max-1-i;j++)
		{
			if(rev[index[j]]>rev[index[j+1]])
			{				
				int temp=index[j];
				index[j]=index[j+1];
				index[j+1]=temp;								
			}
		}
	}	
	cout<<endl;		
	for(int i=0;i<Max;i++)
	{
		cout<<rev[index[i]]<<" "<<in[index[i]]<<endl;
	}	
	return 0;	
}
