#include<iostream>
#define  len   100
using namespace std;
int main()
{   
    char a[len],*p;
    int b[len],i=0,sign=1,num=0;
    cin.getline(a,len);
	for(p=a;*p!='\0';p++)
	{
		if(*p>='0'&&*p<='9')
	  {
		  if(sign)
		{
		 b[i]=*p-48;
		 i++;
	    }   
		 else
		 b[i-1]=b[i-1]*10+*p-48;
		 sign=0;
         num++;
		 cout<<*p;
	  }
      else
      sign=1;
	} 
	cout<<endl;
	cout<<"the number of the integer is:"<<num<<endl;
   
}
