//（6）计算各串逆序数
#include<iostream>
#include<string.h>
using namespace std;
#define MAX 5
int getRev(char *a){
	int n=strlen(a);
	int i,j,count=0;
	for(i=0;i<n;i++){
		for(j=i+1;j<n;j++){
			if(*(a+i)>*(a+j))
				count++;
		}
	}
	return count;
}
int main(){
	char s[MAX][21]={"DAABEC","abc","edbac","shdsilfh","sggiesausgf"};
	int x[MAX],temp1;
	char temp2[21];
	for(int i=0;i<MAX;i++)
		*(x+i)=getRev(*(s+i));
	for(int i=0;i<MAX-1;i++){
		for(int j=0;j<MAX-1-i;j++){
			if(*(x+j)<*(x+j+1)){
				temp1=*(x+j);
				*(x+j)=*(x+j+1);
				*(x+j+1)=temp1;
				strcpy(temp2,*(s+j));
				strcpy(*(s+j),*(s+j+1));
				strcpy(*(s+j+1),temp2);
			}
		}
	}
	for(int i=0;i<MAX;i++)
		cout<<*(s+i)<<':'<<*(x+i)<<endl;	
	return 0;
}
