using namespace std;
int getRev(char a[])
{
	int length = strlen(a);
	int count = 0;
	for(int i=0;i<length;i++)
	{
		for(int j=i+1;j<length;j++)
		{
			if(a[i]>a[j])
				count++;
		}
	}
	return count;
}

void bubbleSort(char * strs[], int n){		
	char *temp;
	for(int i = 0; i < n-1; i++)				
		for(int j = 0; j < n-i-1; j++)
			if(strcmp(strs[j], strs[j+1]) > 0){
				temp = strs[j];				
				strs[j] = strs[j+1];
				strs[j+1] = temp;
			}
}
