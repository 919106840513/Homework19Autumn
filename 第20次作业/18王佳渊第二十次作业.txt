#include<iostream>
#include<string.h>
using namespace std;
int main(void){
	int max, len, n = 0;
	char(*ch)[20] = new char[max][20];
	int* count = new int[max];
	cin >> max;
	cin.get();
	for (int i = 0; i < max; i++){
		cin.getline(ch[i], 20);
		len = strlen(ch[i]);
		for (int j = 0; j < len - 1; j++){
			for (int k = j + 1; k < len; k++){
				if (ch[i][j] > ch[i][k])
					n++;
			}
		}
		count[i] = n;
		n = 0;
	}
	char a[20];
	int b;
	for (int i = 0; i < max - 1; i++){
		for (int j = 0; j < max - 1 - i; j++){
			if (count[j] > count[j + 1]){
				strcpy(a, ch[j]);
				strcpy(ch[j], ch[j + 1]);
				strcpy(ch[j + 1], a);
				b = count[j];
				count[j] = count[j + 1];
				count[j + 1] = b;
			}
		}
	}
	for (int i = 0; i < max; i++)
		cout << ch[i] << "  " << count[i] << endl;
	delete[]ch;
	delete[]count;
	return 0;
}
