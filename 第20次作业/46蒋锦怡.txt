#include<iostream>
#include<string.h>
using namespace std;



int main(void)
{
	int max,len,n=0;
	cin>>max;
	char (*ch)[20]=new char[max][20];
	int *num=new int[max];
	cin.get();
	for(int x=0;x<max;x++)
	{
		cin.getline(ch[x],20);
		len=strlen(ch[x]);
		for(int i=0;i<len-1;i++)
		{
			for(int j=i+1;j<len;j++)
			{
				if(ch[x][i]>ch[x][j])
					n++;	
			}
		}
		num[x]=n;
		n=0;
	}
	char tem1[20];
	int tem2;
	for(int i=0;i<max-1;i++)
	{
		for(int j=0;j<max-1-i;j++)
		{
			if(num[j]>num[j+1])
			{
				strcpy(tem1,ch[j]);
				strcpy(ch[j],ch[j+1]);
				strcpy(ch[j+1],tem1);
				tem2=num[j];
				num[j]=num[j+1];
				num[j+1]=tem2;
			}
		}
	}
	for(int z=0;z<max;z++)
	{
		cout<<ch[z]<<"  "<<num[z]<<endl;
	}
	delete []ch;
	delete []num;
	return 0;
}

struct add{
	char *p=NULL;
	int len=0;
};
int main (){
	int n;
	cin>>n;
	add *a=new add[n];
	for(int i=0;i<n;i++){
	cout<<i+1<<'$';
	cin>>a[i].len;
	a[i].p=new char[a[i].len];
	cin>>a[i].p;
	cout<<endl;
	}

	
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			for(int c=0;*(a[j].p+c);c++){
				if(*(a[j].p+c)>*(a[j].p+c+1)){
					char*d={0};
					strcpy(d,a[j].p+c);
					strcpy(a[j].p+c,a[j].p+c+1);
					strcpy(a[j].p+c+1,d);
					break;
				}
			}
		}
	} 
	for(int i=0;i<n;i++)
	cout<<a[i].p<<endl;
	
	int *s=new int[n];
	for(int i=0;i<n;i++){
		s[i]=0;
	}
	for(int i=0;i<n&&*a[i].p;i++){
		for(int j=i+1;j<n;j++){
		int sum=0;
		bool t=1;
		for(int k=0;k<a[i].len&&k<a[j].len;k++)	{
			if(*(a[i].p+k)!=*(a[j].p+k)) t=0;
			if(t&&a[i].len==a[j].len) {
				sum++;
				strcpy(a[j].p,"/0");
			}
		}
		}
	}
	for(int j=n;j>0;j--){
		for(int i=0;i<n;i++){
			if(s[i]==j){
				cout<<a[i].p<<' '<<s[i]<<endl;
				s[i]=0;
				j++;
			}
		}
	}

}
