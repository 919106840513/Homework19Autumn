 #include<iostream>
#include<string.h>
using namespace std;
int getWordCount(char a[])
{
	int count=1;
	for(int i=0;i<=strlen(a);i++)
	{
		if(a[i]=='\t'||(a[i]==' '&&a[i+1]!=' '))
		count++;
	}
	return count;
}
int main(){
	 char a[100];
	 cin.getline(a,100);
	 cout<<"word count="<<getWordCount(a)<<endl;
	 return 0;
}

 #include<iostream>
#include<string.h>
using namespace std;
#define MAX 3
int getRev(char a[])
{
	int count=0;
	//遍历字符串，并且比较大小，遇到字符串的结尾符'\0'时结束，否则会继续进行！ 
	for(int i=0;a[i]!='\0';i++)
	{
		for(int j=i+1;a[j]!='\0';j++)
		{
			if(a[i]>a[j])
			   count++;
		}
	}
	return count;
}

int main()
{
	//确定输入MAX个字符串，每个字符串字符数不超过20个。 
	char a[MAX][20];
	//得到输入的字符串 
	for(int i=0;i<MAX;i++)
	{
		cin>>a[i];
		cout<<"字符串"<<a[i]<<"的逆序个数是"<<getRev(a[i]) <<endl;
	}
	int t[MAX];
	for(int i=0;i<MAX;i++)
	{
		t[i]=getRev(a[i]);
	}
	//冒泡排序法，将两个有对应关系的数组一起相应排序。 
	for(int j=0;j<MAX;j++)
	{
		for(int i=0;i<MAX-1;i++)//保证其中最大的下标不越界，即i+1<MAX. 
	    {
	         if(t[i]>t[i+1])
	        {
	        	int temp1;char temp2[20];
	        	temp1=t[i];strcpy(temp2,a[i]);
	        	t[i]=t[i+1];strcpy(a[i],a[i+1]);
	        	t[i+1]=temp1;strcpy(a[i+1],temp2);
	     	}
		}
	}
	for(int i=0;i<MAX;i++)
		cout<<a[i]<<":"<<t[i]<<endl;
	return 0;
}
