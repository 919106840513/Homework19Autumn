#include <math.h>

struct Hanshu{
	int x1,x2;
} h;
int f(int a,int b,int c){
	if(b*b-4*a*c>0){
	h.x1=(-b+sqrt(b*b-4*a*c))/(2*a);
	h.x2=(-b-sqrt(b*b-4*a*c))/(2*a);}
	else if(b*b-4*a*c==0)
	h.x1=h.x2=-b/(2*a);
}

	int a,b,c;
	cin>>a>>b>>c;
	f(a,b,c);

}

struct Sanjiao{
	int x,y,z;
} f;
int J(Sanjiao f){
	int s,p;
	if(f.x+f.y==f.z||f.x+f.z==f.y||f.z+f.y==f.x)
	cout<<"not sanjiao";
	else{
		p=(f.x+f.y+f.z)/2;
		s=sqrt(p*(p-f.x)*(p-f.y)*(p-f.z));
		return s;
	}
}

	cin>>o.x>>o.y;
	cin>>f.x>>f.y>>f.z;
