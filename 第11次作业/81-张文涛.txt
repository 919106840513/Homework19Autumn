#include <iostream>
using namespace std;
int y(int n)
{
	int m,sum=0,k;
	m=n;
	while (m)
	{
		k=m%10;
		sum+=k*k*k;
		m/=10;
	}
	if(n==sum)
	return 1;
return 0;
}

int main()
{
  int i,j,n;
  cin>>i>>j;
    for(n=i+1;n<j;n++)
      if(y(n))
  cout<<n<<" ";
}
