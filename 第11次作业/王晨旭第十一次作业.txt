using namespace std;
int main()
{
    int i,j,a,b,c,n;
    cout<<"waterflower number is";
    cin>>i>>j;
    if(i<100||j>999||i>=j)
    cout<<endl;
    else{
	  for(n=i;n<j;n++)
      {
    	a=n/100;//百位 
    	b=(n-100*a)/10;
    	c=n-100*a-10*b;
    	if(n==a*a*a+b*b*b+c*c*c)
    	cout<<n<<endl;
	  }

	}
    return 0;
}
