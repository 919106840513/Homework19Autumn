#include<iostream>
using namespace std;
int length(int n){
int sum=0;
while(n>0){
sum++;
n=n/10;	
}	
return sum;		
}

int cifang(int n){int i=1;int sum=1;
for(i;i<=n;i++)
sum=sum*10;
return sum;	
}



bool huiwen(int n){
int i=1;
int k=length(n)/2;
for(i;i<=k;i++)
if(
(n%cifang(i))/(cifang(i-1))
!=
(n/(cifang(length(n)-i))%10)	
)
break;	
if(i>k)
return 1;
else 
return 0;		
}

int main()
{int n;
cin>>n;
if(huiwen(n)==1)
cout<<"是回文数";
else
cout<<"不是回文数";
}
