#include<iostream>
using namespace std;
char *strcat(char *a, char *b)
{
    char *p = a;
    while(*p) p++;
    
    do
    {
        *p++=*b;
        if(*b == '\0')
        break;
        b++;
    }while(1);

    return a;
}
