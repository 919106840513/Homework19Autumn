#include <iostream>
using namespace std;
void A(char *a,const char *b){
	while(*a!='\0'){
		++a;
	}
	while(*b!='\0'){
		*a++=*b++;
	}
}
int main(){
	char a[]="dsad";
	const char b[]="dgfsafa";
	A(a,b);
	cout<<a;
}
