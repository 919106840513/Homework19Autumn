char *strcat(char *a,char *b)
{
	if (a!=NULL&&b!=NULL)
	{
		char *t=a;
		while (*a)
			a++;
		while (*b!='\0')
		{
			*a=*b;
			a++;
			b++;
		}
		*a='\0';
		return t;
	}	
}

char *trim(char *a)
{
