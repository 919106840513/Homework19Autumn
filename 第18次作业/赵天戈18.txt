#include<iostream>
using namespace std;
void strcat(char *a,char *b)
{
	while(*a!=0)
 	{
 		a++;
	}
	while(*b!=0)
	{
		*a++=*b++;
	}
}
int main()
{
	char a[10],b[10];
	cin.getline(a,10);
	cin.getline(b,10);
	strcat(a,b);
	cout<<a;
}

#include<iostream>
#include<string.h>
using namespace std;
char * trim(char * s)
{
 	if (s==NULL) return NULL;
 	if (strlen(s)==0) return s;
 	char *s1;
 	int j=0;
 	for(int i=0;s[i];i++)
	{
 		if(s[i]!=' ')
		{
 			s1[j]=s[i];
 			j++;
 		}
 	}
 	return s1;
}
int main()
{
 	char s[] = "   ab c  ";
 	cout<<strlen(s);
 	char * s1 = trim(s);
 	cout<<":"<<s1<<":"<<strlen(s1)<<endl;
}
