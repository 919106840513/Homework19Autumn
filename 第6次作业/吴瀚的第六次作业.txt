#include <iostream>
using namespace std;
int f(int n)
{
	return n>1?n*f(n-1):1;
}
 int c(int n,int k)
 {
 return f(n)/(f(k)*f(n-k));
 }
 
 int main(){
  int n,k;
  n=4,k=2; 
  cout<<"C(4,2)="<<c(n,k)<<endl;
  n=6,k=4;
  cout<<"C(6,4)="<<c(n,k)<<endl;
  n=8,k=7;
  cout<<"C(8,7)="<<c(n,k)<<endl;
}
