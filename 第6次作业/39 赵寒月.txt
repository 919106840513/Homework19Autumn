 #include<iostream>
using namespace std;
void C(int n,int k)
{
	int s1=1,s2=1,s3=1;
	for(int j=1;j<=n;j++)
	{
		s1*=j;
	}
	for(int j=1;j<=k;j++)
	{
		s2*=j;
	}
	for(int j=1;j<=n-k;j++)
	{
		s3*=j;
	}
	cout<<"C("<<n<<","<<k<<")="<<s1/s2/s3<<endl;
}
int main()
{
	int n,k;
	cin>>n>>k;
	C(n,k);
	C(4,2);
	C(6,4);
	C(8,7);
	return 0;
	
}
