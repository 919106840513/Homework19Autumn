#include <iostream>
using namespace std;
int jie(int x)
{ int i,count=1;
for(i=1;i<=x;i++)
count*=i;
return count;
	
}
double fun(int x,int y)
{
	double a;
	a=jie(x)/(jie(y)*jie(x-y));
	return a;
}

int main() 
{
	cout<<fun(4,2)<<' '<<fun(6,4)<<' '<<fun(8,7);
	return 0;
}

#include <iostream>
using namespace std;
bool sushu(int n)
{
	int i;
	for(i=2;i*i<=n;i++)
	if(n%i==0)
	return false;
	return true;
	
	
}
int main()
{
	cout<<"孪生素数有：";
    int i,j=0;

	  cout<<endl;
      }

}
