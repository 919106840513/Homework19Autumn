void del(node *head, unsigned m) {
	if (m != 1) {
		node *fore = head, *hind = head;
		while (fore->n != (int)m) {
			hind = fore;
			fore = fore->next;
		}
		node *p = fore;
		hind->next = fore->next;
		delete  p;
	}
	else {
		head = head->next;
		delete head;
	}
}
void destroy(node *head) {
	node *hind=head,*fore=head;
	while(hind->next!=NULL){
		fore=hind->next;
		delete hind;
		hind=fore;
	}
}
